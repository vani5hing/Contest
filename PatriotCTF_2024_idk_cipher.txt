import base64

b64_enc_val = b'QRVWUFdWEUpdXEVGCF8DVEoYEEIBBlEAE0dQAURFD1I='
cipher = base64.b64decode(b64_enc_val).decode()

srt_key = 'secretkey'

res = ''
ser = ''
for i in range(len(cipher) // 2):
    res += chr(ord(cipher[i * 2]) ^ ord(srt_key[i % len(srt_key)]))
    ser += chr(ord(cipher[i * 2 + 1]) ^ ord(srt_key[i % len(srt_key)]))


res = res + ser[::-1]
print(res)
